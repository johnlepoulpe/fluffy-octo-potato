def ratios(nb_data, nb_requests, dim):
    """
    Renvoie la table de donnes des ratios entre les 
    distances aux donnes les plus proches et les plus 
    lointaines de chaque requte alatoire.
    """ 
    t0 = time.clock() # calcule le temps processeur et non le temps physique.
    x = np.random.randn(nb_data,dim) # tableau des donnes
    y = np.random.randn(nb_requests,dim) # tableau des requtes
    z = dist.cdist(y,x,'euclidean') # matrice des distances euclidiennes
    minimums = z.min(1)
    maximums = z.max(1)
    time_taken = time.clock() - t0
    return pd.DataFrame({'Ratios' : (minimums/maximums).tolist(), 
                         'Elapsed' : time_taken, 
                         'N' : nb_data, 
                         'D' : dim})

def ratios_dataframe(nb_data, nb_requests, exp_end):

    results = []

    for i in range(4, exp_end+1):

def random_norm_pos(n, dim):
    """
    Cre n vecteurs rpartis uniformment sur le cadran positif 
    l'enveloppe  de l'hypersphre unit de dimension dim.
    """
    mean = np.zeros(dim)
    cov = np.identity(dim)
    x = np.random.multivariate_normal(mean, cov, n) 
    # x contient n vecteurs distribus selon une loi 
    # normale N(0,1) indpendamment sur chaque axe.
    normed_x = x / np.linalg.norm(x, axis=-1)[:, np.newaxis]
    positiv_x = np.asarray(list(map(np.absolute, normed_x)))
    return positiv_x

Question 4¶

Hub¶

Nous pouvons supposer que le vecteur de dimension D dont toutes les composantes sont gales est un hub.

Cette hypothse vient du raisonnement suivant:
Notons SHP le cadran positif de l'hypersphre unit de dimension D,
Soit v l'unique vecteur de SHP sont toutes les composantes sont gales.
Alors v minimise max{|v-x| | x appartient SHP}.

Anti-Hub¶

Nous pouvons supposer que les vecteurs de la base canonique sont des anti-hub.
Cette hypothse vient du fait qu'ils maximise max{|e-x| | x appartient SHP}.

def normalize(vect):
    """Normalise le vecteur vect."""
    return vect/np.linalg.norm(vect, axis=-1)

def ratios_hypersphere_pos_hub(nb_data, nb_requests, dim):
    """
    Calcul les ratios pour des donnes et requtes, 
    rpartis uniformment sur l'enveloppe sur le cadran 
    positif de l'hypersphre de dimension dim.
    Calcul aussi les rangs du hub et du anti hub.
    """
    t0 = time.clock() 
    
    x = random_norm_pos(nb_data-2, dim) 
    hub = np.ones(dim)
    hub = normalize(hub) 
    anti_hub = np.zeros(dim)
    anti_hub[0] = 1
    x = np.insert(x, 0, hub) # il sera en premire position dans le tableau
    x = np.insert(x, (nb_data-1)*dim, anti_hub) # il sera en dernire position
    x.shape= (nb_data,dim)

    y = random_norm_pos(nb_requests, dim) 
    z = dist.cdist(y,x,'euclidean') 
    
    minimums=z.min(1)
    maximums=z.max(1)
    
    index = np.asarray(list(map(np.argsort, z))) 
    # La ime ligne de index est la liste des indices dans z[i]
    # pour que la liste des [z[i][j] for j in index[i]] soit trie.
    # Ainsi pour trouver les rangs du hub, il faut chercher l'indice 
    # de 0 dans chacune des lignes de index.
    # Pour les anti hub, on fait de mme en cherchant nb_data.
    hub_ranks = np.fromiter(map(np.argmin, index), dtype=np.int)    
    anti_hub_ranks = np.fromiter(map(np.argmax, index), dtype=np.int)

    time_taken = time.clock() - t0
    return pd.DataFrame({'Ratios' : (minimums/maximums).tolist(), 
                         'Hub_ranks' : hub_ranks,
                         'Anti_hub_ranks': anti_hub_ranks, 
                         'Elapsed' : time_taken, 
                         'N' : nb_data, 
                         'D' : dim})

def ratios_moving_point_dataframe(nb_data, nb_requests, exp_end):
    """
    Gnre la table de donnes contenant les rangs des 10 points entre le hub et le anti-hub.
    """
    results = []
    frames = []
    for i in range(4, exp_end+1):
        dim = 2 ** i
        
        hub = np.ones(dim)
        hub = normalize(hub)
        anti_hub = np.zeros(dim)
        anti_hub[0] = 1
    
        x = random_norm_pos(nb_data -1, dim) 
        y = random_norm_pos(nb_requests, dim) 
        z = dist.cdist(y,x,'euclidean') 
    
        for k in range(10):
            moving_point = (1-k/9)*hub + k/9*anti_hub
            # moving_point est alors situ sur le segment [hub, anti_hub]
            # Il est ncessaire de normaliser ce point pour qu'il soit sur l'hypersphre unit.
            moving_point = normalize(moving_point)
            
            distance = dist.cdist(np.array([moving_point]),y,'euclidean')
            current_ratios = np.concatenate((distance, z.T), axis = 0).T

            index = np.asarray(list(map(np.argsort, current_ratios))) 
            moving_point_ranks = np.fromiter(map(np.argmin, index), dtype=np.int)
            
            frames.append(pd.DataFrame({'K' : k,
                                        'D' : dim,
                                        'Ranks' : moving_point_ranks.tolist()}))                         
    
    return pd.concat(frames)

def get_dim_steps(color):
    
    """Renvoie les bonnes valeurs des dimensions  tester selon que les images soient en couleur ou pas."""
    
    if color:
        dim_steps = [2, 4, 8, 10, 13, 16, 20] # valeurs de la question 8
    else:
        dim_steps = [5, 10, 50, 100, 150, 200, 256] # valeurs de la question 7
    return dim_steps

def hellinger(p, q):
    
    """Calcule la distance de Hellinger entre deux vecteurs."""
    
    _SQRT2 = np.sqrt(2) # sqrt(2) with default precision np.float64
    return np.sqrt(np.sum((np.sqrt(p) - np.sqrt(q)) ** 2)) / _SQRT2

def hellinger_mat(A, B):
    
    """Calcule toutes les distances de chaque vecteur de A  chaque vecteur de B."""
    
    A_x, A_y = A.shape
    B_x, B_y = B.shape
        
    C = np.empty([A_x, B_x])
    
    # A et B doivent avoir le mme nombre de colonnes
    for i in range(0, A_x):
        for j in range(0, B_x):
            C[i, j] = hellinger(A[i], B[j]) ** 2

    return C

def im_hists(dim, color = False, truncated = True, hell = False):

    """
    Cre le tableau des histogrammes de toutes les images (N) et les chantillons (M) avec dim points dans chaque.
    color permet d'importer des images en couleur au lieu d'en nuances de gris.
    truncated permet de dfinir la faon dont les histogrammes sont adapts  la dimension requise.
    hell permet d'utiliser la distance de Hellinger au lieu de la distance euclidienne.
    """
    
    # Dclaration de la taille des tableaux des histogrammes
    if color:
        N = np.empty([num_test, dim ** 3])
        M = np.empty([num_orig, dim ** 3])
    else:
        N = np.empty([num_test, dim])
        M = np.empty([num_orig, dim])

    N_i = 0
    M_i = 0

    for image in images:
        
        # Lecture de l'image et cration de l'histogramme
        if color:
            img = cv2.imread("%s%s"%(path, image), cv2.IMREAD_COLOR)
            hist = cv2.calcHist([img],[0, 1, 2], None, [dim, dim, dim], [0, 256, 0, 256, 0, 256])    
        else:
            img = cv2.imread("%s%s"%(path, image), cv2.IMREAD_GRAYSCALE)
            if truncated:
                hist = cv2.calcHist([img],[0],None,[256],[0,256])
            else:
                hist = cv2.calcHist([img],[0],None,[dim],[0,256])

        # Normalisation de l'histogramme
        normhist = hist
        cv2.normalize(hist,normhist,norm_type=cv2.NORM_L2);
        
        # Transformation de l'histogramme en colonne
        if color:
            normhist = normhist.flatten()
        else:
            normhist = np.ravel(normhist)
            # Tronque les donnes si ncessaire
            normhist = normhist[:dim]

        # Ajout de l'histogramme  la matrice pertinente
        if image[:17] == "Copydays_original":
            M[M_i] = normhist
            M_i = M_i + 1
        else:
            N[N_i] = normhist
            N_i = N_i + 1

    return N, M

def im_ratios(dim, color = False, truncated = True, hell = False):
    
    """Calcul des ratios min/max pour chaque chantillon."""
    
    t0 = time.clock() # temps du processeur
    N, M = im_hists(dim, color, truncated, hell)
    
    if hell:
        z = hellinger_mat(M,N) # Distance de Hellinger
    else:
        z = dist.cdist(M,N,'euclidean') # Distance euclidienne
    
    minimums=z.min(1)
    maximums=z.max(1)
    time_taken = time.clock() - t0
    
    return pd.DataFrame({'Ratios' : (minimums/maximums).tolist(), 
                         'Elapsed' : time_taken, 
                         'D' : dim})

def im_ratios_dataframe(color = False, truncated = True, hell = False):

    im_results = []
    im_frames = []

    dim_steps = get_dim_steps(color)
    
    for dim in dim_steps:
        im_frames.append(im_ratios(dim, color, truncated, hell))

    return pd.concat(im_frames)
